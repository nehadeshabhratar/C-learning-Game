A **pointer** is a variable that stores the memory address of another variable. Pointers allow efficient memory management and direct memory access.

ðŸ”¹ Pointer Arithmetic

We can perform arithmetic operations on pointers:

#include 
int main() {
    int arr[3] = {10, 20, 30};
    int *ptr = arr;
    printf("%d ", *ptr);      // Output: 10
    ptr++;                    // Move to next element
    printf("%d ", *ptr);      // Output: 20
    return 0;
}

ðŸ”¹ Pointers and Arrays

Arrays and pointers are closely related.

#include 
int main() {
    int arr[3] = {1, 2, 3};
    int *ptr = arr;
    for (int i = 0; i < 3; i++) {
        printf("%d ", *(ptr + i));  // Access elements using pointer arithmetic
    }
    return 0;
}

ðŸ”¹ Pointers and Functions

ðŸ“Œ 1. Passing Pointers to Functions

#include

    *ptr = 50;
}
int main() {
    int num = 10;
    updateValue(&num);
    printf("Updated value: %d", num);  // Output: 50
    return 0;
}

ðŸ“Œ 2. Returning Pointers from Functions

#include 
int* getPointer(int *ptr) {
    return ptr;
}
int main() {
    int num = 100;
    int *ptr = getPointer(&num);
    printf("%d", *ptr);  // Output: 100
    return 0;
}

ðŸ”¹ Dynamic Memory Allocation

Pointers are used to allocate memory dynamically using **malloc()**, **calloc()**, **realloc()**, and **free()**.

#include 
#include 
int main() {
    int *ptr;
    ptr = (int*) malloc(5 * sizeof(int));  // Allocating memory for 5 integers
    if (ptr == NULL) {
        printf("Memory allocation failed");
        return 1;
    }
    for (int i = 0; i < 5; i++) {
        ptr[i] = i + 1;
    }
    for (int i = 0; i < 5; i++) {
        printf("%d ", ptr[i]);  // Output: 1 2 3 4 5
    }
    free(ptr);  // Free the allocated memory
    return 0;
}
